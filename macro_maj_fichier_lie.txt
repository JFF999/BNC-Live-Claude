' =====================================================================
'  MACRO : ouvrir/mettre a jour le fichier lie Action_2026-c_New.xlsx
'  puis le sauvegarder (rafraichit les liens externes).
'
'  Le dossier Google Drive local est DETECTE selon la langue :
'    - anglais  : C:\Users\<user>\My Drive
'    - francais : C:\Users\<user>\Mon Drive  (ou « Mon disque »)
'    - lecteurs montes G:\ H:\ I:\ en secours
'  (meme logique que chemins_bnc.py cote Python)
' =====================================================================

Function DossierDrive() As String
    ' Renvoie le dossier racine Google Drive existant (FR ou EN), sinon
    ' le candidat par defaut (le message d'erreur affichera ce chemin).
    Dim bases(0 To 3) As String
    Dim noms(0 To 2) As String
    Dim i As Integer, j As Integer
    Dim c As String

    noms(0) = "My Drive"
    noms(1) = "Mon Drive"
    noms(2) = "Mon disque"

    bases(0) = Environ$("USERPROFILE") & "\"
    bases(1) = "G:\"
    bases(2) = "H:\"
    bases(3) = "I:\"

    For i = 0 To 3
        For j = 0 To 2
            c = bases(i) & noms(j)
            On Error Resume Next            ' un lecteur absent (G:) ferait planter Dir
            If Dir(c, vbDirectory) <> "" Then
                On Error GoTo 0
                DossierDrive = c
                Exit Function
            End If
            On Error GoTo 0
        Next j
    Next i

    DossierDrive = bases(0) & noms(0)       ' defaut : My Drive sous le profil
End Function

Sub MettreAJourFichierLie()
    Dim wbLie As Workbook
    Dim chemin As String

    chemin = DossierDrive() & "\Actions\Action_2026-c_New.xlsx"

    If Dir(chemin) = "" Then
        MsgBox "Fichier introuvable : " & chemin, vbExclamation, "Mise a jour"
        Exit Sub
    End If

    Application.ScreenUpdating = False
    Application.DisplayAlerts = False
    On Error GoTo Nettoyage

    Set wbLie = Workbooks.Open( _
        Filename:=chemin, _
        UpdateLinks:=True, _
        ReadOnly:=False, _
        IgnoreReadOnlyRecommended:=True)
    wbLie.Save
    wbLie.Close SaveChanges:=False

Nettoyage:
    Application.ScreenUpdating = True
    Application.DisplayAlerts = True
    If Err.Number <> 0 Then
        MsgBox "Erreur pendant la mise a jour : " & Err.Description, vbExclamation, "Mise a jour"
    End If
End Sub
